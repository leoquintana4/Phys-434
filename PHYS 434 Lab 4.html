# Create Negative Log-Likelihood function
def negative_loglikelihood(param, data):
    loc, scale = param # get loc and scale from parameters
    log_likelihoods = cauchy.logpdf(data, loc=loc, scale=scale) # calculate log likelihoods for each data point
    return -2 * np.sum(log_likelihoods) # return negative log likelihood

# Create chi squared function
def chi_squared(param, data, bin_edges):
    (loc, scale) = param
    observed, _ = np.histogram(data, bins = bin_edges, range= (0,20))
    
    bin_centers = (bin_edges[:-1] + bin_edges[1:]) / 2 
    bin_width = bin_edges[1] - bin_edges[0] 
    
    N_total = len(mass)
    expected = N_total * cauchy.pdf(bin_centers, loc=loc, scale=scale) * bin_width # expected counts per bin
    
    chi_squared_value = np.sum((observed - expected)**2 / expected) # apply chi squared formula
    return chi_squared_value


mass_scan = np.linspace(5, 15, 100) 
chi2_profile = [] #empty list to store chi squared values
ml_best_width = best_fit_width  # fixed best fit width

counter_chi = 0
bin_edges = np.arange(0, 20.5, 0.5)
for m in mass_scan:
    chi2_value = chi_squared([m, ml_best_width], mass, bin_edges) # use chi squared with fixed width
    chi2_value = np.array(chi2_value)
    chi2_profile.append(chi2_value) # add chi squared value to list 
    min_chi2 = np.min(chi2_profile) # find minimum chi squared value 
    delta_chi2_profile = chi2_profile - min_chi2 # calculate delta chi squared 
    counter_chi += 1
    
    
chi2_profile = np.array(chi2_profile) # convert to numpy array

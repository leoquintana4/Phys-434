import numpy as np
import scipy.stats as stats 
import matplotlib.pyplot as plt
# Create random points in square for both x and y
np.random.seed(42)
N = 1000
x = np.random.uniform(0,1, N)
y = np.random.uniform(0,1, N)

# Determine which points are inside and outside the unit circle
inside = x**2 + y**2 < 1
outside = x**2 + y**2 >= 1

# Plot the points and color them red if inside the circle and blue if outside
fig, ax = plt.subplots()
ax.scatter(x[inside], y[inside], color = 'red', s = 1, label = "inside circle")
ax.scatter(x[outside], y[outside], color = 'blue', s = 1, label = "outside circle")
# Draw a unit circle to visualize the boundry
circle = plt.Circle((0, 0), radius=1, edgecolor='black', facecolor='None')
ax.add_patch(circle)
ax.set_xlim([0, 1])
ax.set_ylim([0, 1])
plt.legend()

# Get the sum of points inside and outside the unit circle
inside_count = np.sum(inside)
outside_count = np.sum(outside)


print(f'Number of points inside unit circle" {inside_count})')
print(f'Number of points outside unit circle" {outside_count})')

# Calculate the pi estimate
pi_estimate = (inside_count / N) * 4
# Calculate the uncertainty using binomial std
pi_uncertainty = np.sqrt(inside_count * (1-inside_count / N)) / (N*4)
print(f'Estimated value of pi: {pi_estimate} +/- {pi_uncertainty:.4f}')

Number of points inside unit circle" 778)
Number of points outside unit circle" 222)
Estimated value of pi: 3.112 +/- 0.0033

# Question 2a:
import numpy as np
np.random.seed(42)
N = 10000 # number of samples
sample = [] # create empty list to store samples
for i in range(N): # loop over 
    x = np.random.uniform(-1, 1) # generator x_i
    y = np.random.uniform(0, 1) # generator y_i
    if y < x**2: # check if y_i is less than f(x_i) 
        sample.append(x) # reject if y_i > f(x_i)
        
sample = np.array(sample) # convert list to array
        
print(f'Samples values that have the y_i < f(x_i) condition true: {sample}')

# Question 2b:
np.random.seed(42)
N = 10000 # number of samples
y2 =np.random.uniform(-1,1, N) # generate uniform distribution between -1 and 1    
x_cdf = np.cbrt(3*y2) # apply inverse transformation function
print(f'Samples values from inverse transformation method: {y2}')

 # Plot both methods together 
ax, fig = plt.subplots()
plt.hist(3/2 * sample, bins = 40, density = True, alpha = 0.5, label = 'Rejection method', color = 'blue')
plt.hist(x_cdf, bins = 40, density = True, alpha = 0.5, label = 'Inverse transformation method', color = 'red')

# Plot the target distribution f(x) = x**2
x = np.linspace(-1, 1, 1000)
plt.plot(3/2 * x, x**2, label = 'Target Distribution', color = 'black')

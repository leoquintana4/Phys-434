np.random.seed(4)
N = 10
p = 0.02
M = 10000
# create the binomial distribution
binomial_samples = np.random.binomial(N, p, M)
# create the normal distribution
normal_samples = np.random.normal(N*p, np.sqrt(N*p*(1-p)), M)
# create the poisson distribution
poisson_samples = np.random.poisson(N*p, M)

plt.figure(figsize=(10, 6))
plt.hist(binomial_samples, bins=25, density=True, alpha=0.6, color='b', label='Binomial Distribution') 
plt.hist(normal_samples, bins=25, density=True, alpha=0.6, color='r', label='Normal Distribution')
plt.hist(poisson_samples, bins=25, density=True, alpha=0.6, color='g', label='Poisson Distribution')

# Create table of means, sigma, skewness, and kurtosis with pandas
data = {
    'Distribution': ['Binomial', 'Normal', 'Poisson'],
    'Mean': [np.mean(binomial_samples), np.mean(normal_samples), np.mean(poisson_samples)],
    'Sigma': [np.std(binomial_samples), np.std(normal_samples), np.std(poisson_samples)],
    'Skewness': [stats.skew(binomial_samples), stats.skew(normal_samples), stats.skew(poisson_samples)],
    'Kurtosis': [stats.kurtosis(binomial_samples), stats.kurtosis(normal_samples), stats.kurtosis(poisson_samples)]
}
pd_table = pd.DataFrame(data)
print(pd_table)
